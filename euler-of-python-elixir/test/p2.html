defmodule P2 do
  def binet_fib(n) do
    s5 = :math.sqrt(5)
    pplus = (1+s5)/2
    pminu = (1-s5)/2
    1/s5*(:math.pow(pplus,n)-:math.pow(pminu,n)) |> round
  end
end
